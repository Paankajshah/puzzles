I used nodejs to create Web api and 
followings docker commands are used in this puzzle to create image and then pushed to docker hub

FROM alpine:latest
RUN apk add --no-cache nodejs npm
COPY . app/
WORKDIR /app
RUN npm install
EXPOSE 8080 

ENTRYPOINT [ "node" ]
CMD [ "index.js" ]
